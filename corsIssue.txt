var corsPolicyName = "AllowAngular";
builder.Services.AddCors(options =>
{
    options.AddPolicy(name : corsPolicyName, policy =>
    {
        policy.AllowAnyOrigin()
              .AllowAnyHeader()
              .AllowAnyMethod();
    });
});
